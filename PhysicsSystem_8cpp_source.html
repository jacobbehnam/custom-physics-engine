43std::optional<std::vector<ObjectSnapshot>> Physics::PhysicsSystem::fetchLatestSnapshot(float renderSimTime) {
44 if (!snapshotReady.load(std::memory_order_acquire))
45 return std::nullopt;
46
47 std::lock_guard<std::mutex> lk(snapshotMutex);
48
49 // first call ever: no previous to interpolate against
50 if (previousSnapshots.empty()) {
51 previousSnapshots = currentSnapshots;
52 return currentSnapshots;
53 }
54
55 // grab the two timestamps (they’re all the same within each frame)
56 float t0 = previousSnapshots[0].time;
57 float t1 = currentSnapshots[0].time;
58
59 // outside the bracket → just clamp
60 if (renderSimTime <= t0) {
61 previousSnapshots = currentSnapshots;
62 return previousSnapshots;
63 }
64 if (renderSimTime >= t1) {
65 previousSnapshots = currentSnapshots;
66 return currentSnapshots;
67 }
68
69 // compute blend factor α in (0,1)
70 float alpha = (renderSimTime - t0) / (t1 - t0);
71
72 // interpolate each ObjectSnapshot
73 std::vector<ObjectSnapshot> out;
74 out.reserve(currentSnapshots.size());
75 for (size_t i = 0; i < currentSnapshots.size(); ++i) {
76 const auto &A = previousSnapshots[i];
77 const auto &B = currentSnapshots[i];
78
79 ObjectSnapshot C;
80 C.body = A.body; // same pointer/ID
81 C.time = renderSimTime; // stamped with the render time
82 C.position = glm::mix(A.position, B.position, alpha);
83 C.velocity = glm::mix(A.velocity, B.velocity, alpha);
84
85 out.push_back(C);
86 }
87
88 // slide window: next time, previous = what we just returned
89 previousSnapshots = currentSnapshots;
90 return out;

95

101 lastTime = now;
102
103 if (!physicsEnabled.load()) {
104 std::this_thread::sleep_for(std::chrono::milliseconds(1));
105 accumulator = 0.0f;
106 lastTime = std::chrono::high_resolution_clock::now();

108 }
109
110 accumulator += frameTime * getSimSpeed();

115 while (accumulator >= dt) {
116 if (step(dt)) {
117 accumulator = 0.0f;
118 break;
119 }
120 accumulator -= dt;
121 }

123 }
124

126 localSnaps.reserve(localBodies.size());
127 {
128 std::lock_guard<std::mutex> lk(snapshotMutex);
129 for (auto* body : localBodies) {

131 }
132
133 currentSnapshots = std::move(localSnaps);
134 }

156void Physics::PhysicsSystem::advancePhysics(float dt) {
157 float targetTime = stepCount.load() * dt + dt; // the time we will be at after this step
158 for (auto body : bodies) {
159 std::unique_lock<std::mutex> guard = body->lockState();
160 body->setForce("Normal", glm::vec3(0.0f), BodyLock::NOLOCK);
161 body->setForce("Gravity", body->getMass(BodyLock::NOLOCK) * getGlobalAcceleration(), BodyLock::NOLOCK);
162
163 if (body->getIsStatic(BodyLock::NOLOCK))
164 continue;
165
166 if (simTime == 0.0f) {
167 body->recordFrame(0.0f, BodyLock::NOLOCK);
168
169 if (resetState.find(body) == resetState.end()) {
170 resetState[body] = body->getAllFrames(BodyLock::NOLOCK).front();
171 }
172 }
173
174 body->step(dt, BodyLock::NOLOCK);
175 body->recordFrame(targetTime, BodyLock::NOLOCK);
176 }
177
178 for (int i = 0; i < bodies.size(); ++i) {
179 for (int j = i + 1; j < bodies.size(); ++j) {
180 PhysicsBody* a = bodies[i];
181 PhysicsBody* b = bodies[j];
182
183 if (a->getIsStatic(BodyLock::LOCK) && b->getIsStatic(BodyLock::LOCK)) continue;
184
185 if (a->collidesWith(*b)) {
186 a->resolveCollisionWith(*b);
187 }
188 }
189 }
190 stepCount++;
191 simTime = targetTime;
192}

194bool Physics::PhysicsSystem::step(float dt) {
195 if (solver && solver->stepFrame()) {
196 std::cout << "Solver Converged!" << std::endl;
197
198 float finalDuration = this->simTime - (dt * 0.5f); // ensures no extra step is made due to floating point errors
199
200 for (auto body : bodies) {
201 const auto& frames = body->getAllFrames(BodyLock::LOCK);
202 if (!frames.empty()) {
203 // The first frame is always t=0 for the current trajectory
204 resetState[body] = frames.front();
205 }
206 }
207 reset();
208
209 float bakeTimer = 0.0f;
210
211 while (bakeTimer < finalDuration) {
212 advancePhysics(dt);
213 bakeTimer += dt;
214 }
215
216 solver = nullptr;
217 physicsEnabled = false;
218 return true;
219 }
220
221 advancePhysics(dt);
222 return false;
223}
