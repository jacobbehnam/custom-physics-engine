64bool Gizmo::rayIntersection(glm::vec3 rayOrigin, glm::vec3 rayDir, float &outDistance){
65 Physics::Bounding::AABB localAABB = getMesh()->getLocalAABB();
66 IHandle* hitHandle = nullptr;
67 float closestT = std::numeric_limits<float>::infinity();
68 for (IHandle* handle : handles) {
69 float outT;

71 if (outT < closestT) {
72 closestT = outT;
73 hitHandle = handle;

75 }
76 if (hitHandle)
77 outDistance = closestT;
78 }
79 if (!activeHandle || !isDragging)
