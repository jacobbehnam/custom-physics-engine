147void SceneManager::setObjectName(SceneObject *obj, const std::string &newName) {
148 assert(obj);
149
150 const std::string& oldName = obj->getName();
151 if (oldName == newName)
152 return;
153
154 assert(isNameUnique(newName, obj));
155
156 if (!oldName.empty()) {

159 usedNames.erase(it);
160 }
161 }
162
163 obj->setName(newName);
164 usedNames[newName] = obj;

169std::string SceneManager::makeUniqueName(const std::string &baseName) const {
170 if (usedNames.find(baseName) == usedNames.end())
171 return baseName;
172
173 int index = 1;
174 while (true) {
175 std::string candidate = baseName + " (" + std::to_string(index) + ")";
176 if (usedNames.find(candidate) == usedNames.end())
177 return candidate;
178 index++;
179 }

216void SceneManager::handleMouseButton(Qt::MouseButton button, QEvent::Type type, Qt::KeyboardModifiers mods) {
217 const bool isPress = (type == QEvent::MouseButtonPress);
218 const bool isRelease = (type == QEvent::MouseButtonRelease);
219 Camera* camera = scene->getCamera();
220
221 if (button == Qt::RightButton) {
222 if (isPress) {
223 rightClickStartDir = camera->front;
224
225 if (!window->isMouseCaptured()) {
226 window->setMouseCaptured(true);
227 camera->resetMouse();
228 }
229 }
230 else if (isRelease && window->isMouseCaptured()) {
231 window->setMouseCaptured(false);
232 camera->resetMouse();
233
234 if (glm::distance(camera->front, rightClickStartDir) < 0.05f) {
235 Math::Ray ray = getMouseRay();
236 IPickable* hit = Math::findFirstHit(pickableObjects, ray, currentGizmo.get())->object;
237
238 if (auto* sceneObj = dynamic_cast<SceneObject*>(hit)) {
239 emit contextMenuRequested(QCursor::pos(), sceneObj);
240 }
241 }
242 }
243 }
244
245 if (button == Qt::LeftButton) {
246 Math::Ray ray = getMouseRay();
247 auto hit = Math::findFirstHit(pickableObjects, ray, currentGizmo.get());
248 IPickable* clickedObject = hit ? hit->object : nullptr;
249 bool clickedCurrentGizmo = clickedObject && currentGizmo && (clickedObject->getObjectID() == currentGizmo->getObjectID());
250
251 if (isPress) {
252 // Clear selection if we click something new that isn't the current gizmo
253 if (!clickedObject || (!selectedIDs.empty() && !clickedCurrentGizmo)) {
254 setSelectFor(nullptr);
255 }
256
257 if (!clickedObject)
258 return;
259
260 clickedObject->handleClick(ray, hit->distance);
261 selectedIDs.insert(clickedObject->getObjectID());
262 if (!clickedCurrentGizmo) {
263 emit selectedItem(dynamic_cast<SceneObject*>(clickedObject));
264 }
265 } else if (isRelease && currentGizmo) {
266 currentGizmo->handleRelease();
267 selectedIDs.erase(currentGizmo->getObjectID());
268 }
269 }

272void SceneManager::processHeldKeys(const QSet<int> &heldKeys, float dt) {
273 Camera* camera = scene->getCamera();
274
275 if (heldKeys.contains(Qt::Key_Escape))
276 qApp->quit();
277
278 static const std::unordered_map<int, GizmoType> keyToGizmoType = {
279 {Qt::Key_T, GizmoType::TRANSLATE},
280 {Qt::Key_R, GizmoType::ROTATE},
281 {Qt::Key_E, GizmoType::SCALE}
282 };
283
284 for (auto& [key, type] : keyToGizmoType) {
285 if (heldKeys.contains(key)) {
286 GizmoType oldType = selectedGizmoType;
287 selectedGizmoType = type;
288 if (currentGizmo && oldType != selectedGizmoType)
289 setGizmoFor(currentGizmo->getTarget(), true);
290 break;
291 }
292 }
293
294 if (currentGizmo && currentGizmo->getIsDragging()) {
295 hoveredIDs.insert(currentGizmo->getObjectID());
296 Math::Ray ray = getMouseRay();
297 currentGizmo->handleDrag(ray);
298 }
299
300 if (heldKeys.contains(Qt::Key_A))
301 camera->processKeyboard(Movement::LEFT, dt);
302 if (heldKeys.contains(Qt::Key_D))
303 camera->processKeyboard(Movement::RIGHT, dt);
304 if (heldKeys.contains(Qt::Key_W))
305 camera->processKeyboard(Movement::FORWARD, dt);
306 if (heldKeys.contains(Qt::Key_S))
307 camera->processKeyboard(Movement::BACKWARD, dt);
308
309 // TODO: Z,X,P,O should be processed by press once
310 // It currently triggers every frame (caused bugs)
311 if (heldKeys.contains(Qt::Key_Z)) {
312 physicsSystem->enablePhysics();
313 }
314 if (heldKeys.contains(Qt::Key_X)) {
315 physicsSystem->disablePhysics();
316 }
317 if (heldKeys.contains(Qt::Key_P)) {
318 if (saveScene("scene.json"))
319 std::cout << "Save Success!" << std::endl;
320 else
321 std::cout << "Save Failed!" << std::endl;
322 }
323 if (heldKeys.contains(Qt::Key_O)) {
324 if (loadScene("scene.json"))
325 std::cout << "Load Success!" << std::endl;
326 else
327 std::cout << "Load Failed!" << std::endl;
328 }
329}
