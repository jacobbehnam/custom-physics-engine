133void SceneManager::setObjectName(SceneObject *obj, const std::string &newName) {
134 assert(obj);
135
136 const std::string& oldName = obj->getName();
137 if (oldName == newName)
138 return;
139
140 assert(isNameUnique(newName, obj));
141
142 if (!oldName.empty()) {

144 if (it != usedNames.end() && it->second == obj) {
145 usedNames.erase(it);
146 }
147 }
148
149 obj->setName(newName);
150 usedNames[newName] = obj;
151

155std::string SceneManager::makeUniqueName(const std::string &baseName) const {
156 if (usedNames.find(baseName) == usedNames.end())
157 return baseName;
158

163 return candidate;
164 index++;

202void SceneManager::handleMouseButton(Qt::MouseButton button, QEvent::Type type, Qt::KeyboardModifiers mods) {
203 const bool isPress = (type == QEvent::MouseButtonPress);
204 const bool isRelease = (type == QEvent::MouseButtonRelease);
205 Camera* camera = scene->getCamera();
206
207 if (button == Qt::RightButton) {
208 if (isPress) {
209 rightClickStartDir = camera->front;
210
211 if (!window->isMouseCaptured()) {
212 window->setMouseCaptured(true);
213 camera->resetMouse();
214 }
215 }
216 else if (isRelease && window->isMouseCaptured()) {
217 window->setMouseCaptured(false);
218 camera->resetMouse();
219
220 if (glm::distance(camera->front, rightClickStartDir) < 0.05f) {
221 MathUtils::Ray ray = getMouseRay();
222 IPickable* hit = MathUtils::findFirstHit(pickableObjects, ray, currentGizmo.get())->object;
223
224 if (auto* sceneObj = dynamic_cast<SceneObject*>(hit)) {
225 emit contextMenuRequested(QCursor::pos(), sceneObj);
226 }
227 }
228 }
229 }
230
231 if (button == Qt::LeftButton) {
232 MathUtils::Ray ray = getMouseRay();
233 auto hit = MathUtils::findFirstHit(pickableObjects, ray, currentGizmo.get());
234 IPickable* clickedObject = hit ? hit->object : nullptr;
235 bool clickedCurrentGizmo = clickedObject && currentGizmo && (clickedObject->getObjectID() == currentGizmo->getObjectID());
236
237 if (isPress) {
238 // Clear selection if we click something new that isn't the current gizmo
239 if (!clickedObject || (!selectedIDs.empty() && !clickedCurrentGizmo)) {
240 setSelectFor(nullptr);
241 }
242
243 if (!clickedObject)
244 return;
245
246 clickedObject->handleClick(ray.origin, ray.dir, hit->distance);
247 selectedIDs.insert(clickedObject->getObjectID());
248 if (!clickedCurrentGizmo) {
249 emit selectedItem(dynamic_cast<SceneObject*>(clickedObject));
250 }
251 } else if (isRelease && currentGizmo) {
252 currentGizmo->handleRelease();
253 selectedIDs.erase(currentGizmo->getObjectID());
254 }
255 }

258void SceneManager::processHeldKeys(const QSet<int> &heldKeys, float dt) {
259 Camera* camera = scene->getCamera();
260
261 if (heldKeys.contains(Qt::Key_Escape))
262 qApp->quit();
263
264 static const std::unordered_map<int, GizmoType> keyToGizmoType = {
265 {Qt::Key_T, GizmoType::TRANSLATE},
266 {Qt::Key_R, GizmoType::ROTATE},
267 {Qt::Key_E, GizmoType::SCALE}
268 };
269
270 for (auto& [key, type] : keyToGizmoType) {
271 if (heldKeys.contains(key)) {
272 GizmoType oldType = selectedGizmoType;
273 selectedGizmoType = type;
274 if (currentGizmo && oldType != selectedGizmoType)
275 setGizmoFor(currentGizmo->getTarget(), true);
276 break;
277 }
278 }
279
280 if (currentGizmo && currentGizmo->getIsDragging()) {
281 hoveredIDs.insert(currentGizmo->getActiveHandle()->getObjectID());
282 MathUtils::Ray ray = getMouseRay();
283 currentGizmo->handleDrag(ray.origin, ray.dir);
284 }
285
286 if (heldKeys.contains(Qt::Key_A))
287 camera->processKeyboard(Movement::LEFT, dt);
288 if (heldKeys.contains(Qt::Key_D))
289 camera->processKeyboard(Movement::RIGHT, dt);
290 if (heldKeys.contains(Qt::Key_W))
291 camera->processKeyboard(Movement::FORWARD, dt);
292 if (heldKeys.contains(Qt::Key_S))
293 camera->processKeyboard(Movement::BACKWARD, dt);
294
295 if (heldKeys.contains(Qt::Key_Z)) {
296 physicsSystem->enablePhysics();
297 }
298 if (heldKeys.contains(Qt::Key_X)) {
299 physicsSystem->disablePhysics();
300 }
301
302 if (heldKeys.contains(Qt::Key_P)) {
303 if (saveScene("scene.json"))
304 std::cout << "Save Success!" << std::endl;
305 else
306 std::cout << "Save Failed!" << std::endl;
307 }
308 if (heldKeys.contains(Qt::Key_O)) {
309 if (loadScene("scene.json"))
310 std::cout << "Load Success!" << std::endl;
311 else
312 std::cout << "Load Failed!" << std::endl;
313 }
