106glm::vec3 Octree::computeForce(Physics::PhysicsBody* body, float G) {
107 if (nodes.empty() || body == nullptr) {
108 return glm::vec3(0.0f);
109 }
110

112 glm::vec3 bodyPos = body->getPosition(BodyLock::NOLOCK);
113 float bodyMass = body->getMass(BodyLock::NOLOCK);
114
115 // Use vector incase overflow
116 // I was planning for 512 elements but not sure
117 std::vector<NodeIndex> stack;
118 stack.reserve(512);
119 stack.push_back(NodeIndex::rootIndex());
120
121 while (!stack.empty()) {
122 NodeIndex currentIdx = stack.back();
123 stack.pop_back();
124 const OctreeNode& node = nodes[currentIdx.val];
125
126 // Empty region
127 if (node.totalMass == 0.0f) continue;
128

130 float distSq = glm::dot(dist, dist);
131 float softeningDistSq = distSq + Constants::SOFTENING_SQ;
132 float widthSq = node.halfSize * node.halfSize * 4.0f;
133
134 // Only single body or sufficiently far away, treat as point mass
135 if (node.isLeaf() || widthSq < Constants::THETA_SQ * distSq) {
136 if (node.body == body) continue; // Skip self
137
138 float invDist = 1.0f / sqrt(softeningDistSq);
139 float invDist3 = invDist * invDist * invDist;
140
141 float force = (G * bodyMass * node.totalMass) * invDist3;
142 totalForce += force * dist;
143 } else {
144 // Add valid children to stack
145 std::uint8_t childMask = node.childMask;
146 while (childMask) {
147 int childOctant = std::countr_zero(childMask);
148 stack.push_back(node.children[childOctant]);
149 childMask &= (childMask - 1);
150 }
151 }
152 }
