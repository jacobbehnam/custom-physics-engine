123void MainWindow::showObjectContextMenu(const QPoint &pos, SceneObject *obj) {

129 connect(solveAction, &QAction::triggered, [this, obj]() {
130 Physics::PhysicsBody* body = obj->getPhysicsBody();
131
132 if (body) {
133 const ProblemRouter *router = sceneManager->physicsSystem->getRouter();
134 // Create and show the dialog
135 SolverDialog dialog(router, body, this);
136 if (dialog.exec() == QDialog::Accepted) {
137 auto knowns = dialog.getCollectedKnowns();
138 std::string unknown = dialog.getTargetUnknown();
139
140 sceneManager->physicsSystem->solveProblem(body, knowns, unknown);
141 }
142 } else {
143 qDebug() << "Selected object has no physics body attached.";
144 }
145 });
146
147 contextMenu.exec(pos);
