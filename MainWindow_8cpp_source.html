235void MainWindow::showObjectContextMenu(const QPoint &pos, SceneObject *obj) {
236 if (!obj) return;
237
238 QMenu contextMenu;
239 QAction* solveAction = contextMenu.addAction("Open Solver...");
240
241 connect(solveAction, &QAction::triggered, [this, obj]() {
242 Physics::PhysicsBody* body = obj->getPhysicsBody();
243
244 if (body) {
245 const ProblemRouter *router = sceneManager->physicsSystem->getRouter();
246 // Create and show the dialog
247 SolverDialog dialog(router, body, this);
248 if (dialog.exec() == QDialog::Accepted) {
249 auto knowns = dialog.getCollectedKnowns();
250 std::string unknown = dialog.getTargetUnknown();
251
252 sceneManager->physicsSystem->solveProblem(body, knowns, unknown);
253 }
254 } else {
255 qDebug() << "Selected object has no physics body attached.";
256 }
257 });
258
259 contextMenu.exec(pos);

261void MainWindow::onHierarchySelectionChanged(SceneObject *previous, SceneObject *current) {
262 if (previous) {
263 sceneManager->setSelectFor(previous, false);
264 }
265 if (current) {
266 sceneManager->setSelectFor(current, true);
267 sceneManager->setGizmoFor(current, true);
268 inspector->loadObject(current);
269
270 if (auto* selectedBody = current->getPhysicsBody()) {
271 selectedBody->withFrames(BodyLock::LOCK, [this](const std::vector<ObjectSnapshot>& snapshots) {
272 snapshotModel->setSnapshots(snapshots);
273 frameGraphPanel->loadSnapshots(snapshots);
274 });
275 } else {
276 snapshotModel->setSnapshots({});
277 frameGraphPanel->clear();
278 }
279 } else {
280 inspector->unloadObject();
281 snapshotModel->setSnapshots({});
282 frameGraphPanel->clear();
283 }
284}
