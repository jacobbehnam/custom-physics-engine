100bool SceneObject::intersectsMesh(const glm::vec3 &orig, const glm::vec3 &dir, float &outT) const {
101 const std::vector<Vertex>& verts = mesh->getVertices();
102 std::vector<glm::vec3> vertPositions;
103 vertPositions.reserve(verts.size());
104 for (auto vert : verts) {
105 vertPositions.push_back({vert.pos});
106 }
107 const size_t triCount = verts.size() / 3;
108
109 const std::vector<unsigned int>& indices = mesh->getIndices();
110 auto *glFuncs = QOpenGLVersionFunctionsFactory::get<QOpenGLFunctions_4_5_Core>(QOpenGLContext::currentContext());
111 ComputeShader compute("assets/shaders/meshIntersection.comp", glFuncs);
112 (void) compute.createSSBO(vertPositions.data(), vertPositions.size() * sizeof(glm::vec3), 0);
113 (void) compute.createSSBO(indices.data(), indices.size() * sizeof(unsigned int), 1);
114 std::vector<float> initDists(triCount, -1.0f);
115 unsigned int distancesSSBO = compute.createSSBO(initDists.data(), initDists.size() * sizeof(float), 2);
116 compute.use();
117 compute.setVec3("rayOrig", orig);
118 compute.setVec3("rayDir", dir);
119 compute.setMat4("modelMatrix", getModelMatrix());
120
121 constexpr unsigned int groupSize = 64;
122 unsigned int groups = (triCount + groupSize - 1) / groupSize;
123 compute.dispatch(groups);
124
125 auto distances = compute.readSSBO<float>(distancesSSBO, triCount);
126 auto minIt = std::min_element(distances.begin(), distances.end());
127 if (minIt == distances.end()) {
128 return false;
129 }
130 outT = *minIt;
131 return true;
132}
