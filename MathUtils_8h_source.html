65 inline std::optional<float> intersectTriangle(
66 const Ray& ray,
67 const glm::vec3& v0,
68 const glm::vec3& v1,
69 const glm::vec3& v2
70 ) {
71 constexpr float EPSILON = std::numeric_limits<float>::epsilon();
72
73 glm::vec3 edge1 = v1 - v0;
74 glm::vec3 edge2 = v2 - v0;
75
76 glm::vec3 pvec = glm::cross(ray.dir, edge2);
77 float det = glm::dot(edge1, pvec);
78
79 // If the determinant is near zero, ray lies in plane of triangle or is backfacing
80 if (det > -EPSILON && det < EPSILON) return std::nullopt;
81
82 float invDet = 1.0f / det;
83
84 glm::vec3 tvec = ray.origin - v0;
85 float u = glm::dot(tvec, pvec) * invDet;
86 if (u < 0.0f || u > 1.0f) return std::nullopt;
87
88 glm::vec3 qvec = glm::cross(tvec, edge1);
89 float v = glm::dot(ray.dir, qvec) * invDet;
90 if (v < 0.0f || u + v > 1.0f) return std::nullopt;
91
92 // compute t to find out where the intersection point is on the line
93 float t = glm::dot(edge2, qvec) * invDet;
94 if (t < EPSILON) return std::nullopt; // intersection is behind the ray origin
95
96 return t;
97 }

128 inline glm::vec3 screenToWorldRayDirection(
129 double mouseX,
130 double mouseY,
131 int fbWidth,
132 int fbHeight,
133 const glm::mat4 &view,
134 const glm::mat4 &projection
135 ) {
136 float x = (2.0f * mouseX) / fbWidth - 1.0f;
137 float y = 1.0f - (2.0f * mouseY) / fbHeight;
138
139 glm::vec4 clip = glm::vec4(x, y, -1.0f, 1.0f);
140 glm::mat4 invProj = glm::inverse(projection);
141 glm::vec4 eye = invProj * clip;
142 eye.z = -1.0f;
143 eye.w = 0.0f;
144
145 glm::mat4 invView = glm::inverse(view);
146 glm::vec4 worldDir4 = invView * eye;
147
148 return glm::normalize(glm::vec3(worldDir4));
149 }

182 inline std::optional<HitResult> findFirstHit(
183 const std::vector<IPickable*>& objects,

185 IPickable* priority = nullptr

187 float distance = std::numeric_limits<float>::infinity();
188 IPickable* best = nullptr;
189
190 for (IPickable* obj : objects) {
191 float t;
192 if (!obj->rayIntersection(ray.origin, ray.dir, t))
193 continue;
194
195 // If this is the priority object, take it immediately
196 if (obj == priority) {
197 distance = t;
198 return HitResult{obj, distance};
199 }
200
201 if (t < distance) {
202 distance = t;
203 best = obj;
204 }
205 }
206
207 return HitResult{best, distance};
208 }
