21 int nearestIndexByX(const std::vector<QPointF>& pts, qreal mx) {
22 if (pts.empty()) return -1;
23 const auto it = std::lower_bound(pts.begin(), pts.end(), mx,
24 [](const QPointF& p, qreal x) { return p.x() < x; });
25 if (it == pts.begin()) {
26 return 0;
27 }
28 if (it == pts.end()) {
29 return static_cast<int>(pts.size() - 1);
30 }
31 const int i1 = static_cast<int>(it - pts.begin());
32 const int i0 = i1 - 1;
33 const qreal d0 = std::abs(pts[static_cast<size_t>(i0)].x() - mx);
34 const qreal d1 = std::abs(pts[static_cast<size_t>(i1)].x() - mx);
35 return d0 <= d1 ? i0 : i1;
36 }

173void FrameGraphCanvas::rebuildPoints() {
174 graphPoints.clear();
175 hoverIndex = -1;
176 if (!framesRef || framesRef->empty()) return;
177 const QRect rect = plotRect();
178 if (rect.width() <= 1 || rect.height() <= 1) return;
179
180 const int m = static_cast<int>(currentMetric);
181 if (m < 0 || m >= static_cast<int>(kPlottableMetricCount)) return;
182 const float minTime = tMin;
183 const float maxTime = tMax;
184 const float minValue = valueMinMaxPerMetric[static_cast<size_t>(m)].first;
185 const float maxValue = valueMinMaxPerMetric[static_cast<size_t>(m)].second;
186
187 graphPoints.reserve(framesRef->size());
188 const float invTime = maxTime > minTime ? 1.0f / (maxTime - minTime) : 0.0f;
189 const float invValue = maxValue > minValue ? 1.0f / (maxValue - minValue) : 0.0f;
190
191 for (const auto& frame : *framesRef) {
192 const float timeAlpha = (frame.time - minTime) * invTime;
193 const float valueAlpha = (objectSnapshotValue(currentMetric, frame) - minValue) * invValue;
194 const qreal x = rect.left() + static_cast<qreal>(timeAlpha) * rect.width();
195 const qreal y = rect.bottom() - static_cast<qreal>(valueAlpha) * rect.height();
196 graphPoints.emplace_back(x, y);
197 }
