42std::unique_ptr<ISolver> ProblemRouter::makeSolver(Physics::PhysicsBody* body, const std::unordered_map<std::string, double> &knowns, const std::string &unknown) const {
43 auto it = solverMap.find(unknown);
44 if (it == solverMap.end()) {
45 std::cerr << "No solver registered for unknown: " << unknown << std::endl;
46 return nullptr;
47 }
48
49 for (const auto& entry : it->second) {
50 if (areRequirementsMet(entry.requiredKeys, knowns)) {
51 return entry.factory(body, knowns);
52 }
53 }
54
55 std::cerr << "Insufficient knowns provided for unknown: " << unknown << std::endl;
